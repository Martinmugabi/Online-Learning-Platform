Customer Segmentation Analysis

In [1]: import pandas as pd
In [2]: data = pd.read_csv('customer_data.csv')
In [3]: data.head() 
   age  income  spending_score
0   19      15             39
1   21      15             81
2   20      16              6
In [4]: from sklearn.cluster import KMeans
In [5]: kmeans = KMeans(n_clusters=5)
In [6]: clusters = kmeans.fit_predict(data[['income','spending_score']])
In [7]: import matplotlib.pyplot as plt
In [8]: plt.scatter(data['income'], data['spending_score'], c=clusters)
In [9]: plt.show()
